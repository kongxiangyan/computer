```bash
# 查看主机名
localhost -f
# 指定主机名
sudo hostnamectl set-hostname vice.thoughts.vip phabricator.thoughts.vip

# -p 递归创建父级目录
mkdir 

# 只能删除目录不能删除文件，且不能删除非空目录
# -p 递归删除父级目录
rmdir 

# 删除目录或文件
# -f 强制执行删除动作
# -i 互动模式【是否在执行前向用户弹出Confirm】
# -r
rm -rf ../


# 查找命令的绝对路径
which [command] # which ls

# 设置命令的别名
alias [alias]=[command name] # alias rm='rm -i'

```
